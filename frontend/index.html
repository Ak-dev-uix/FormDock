// Example ID
const FORM_ID = "NnM5mtph9qFMOcuIGulff";
const API_URL = `http://fordock.vercel.aap/api/forms/${FORM_ID}/submissions`;

async function fetchSubmissions() {
  try {
    const res = await fetch(API_URL);
    const data = await res.json();

    const list = document.getElementById("submissions-list");
    list.innerHTML = "";

    if (!data || data.length === 0) {
      list.innerHTML = `No submissions yet. Try filling out your form first!`;
      return;
    }

    data.forEach((submission, i) => {
      const box = document.createElement("div");

      const responses = submission.responses || {};
      const responseKeys = Object.keys(responses);
      let responseHTML = '';

      responseKeys.forEach(key => {
        responseHTML += `${key}: ${responses[key]}\n`;
      });

      box.innerText = `
Submission #${i + 1}
ID: ${submission.id}

${responseHTML}

Submitted on: ${new Date(submission.submittedAt || submission.createdAt).toLocaleString()}
      `;
      list.appendChild(box);
    });
  } catch (err) {
    document.getElementById("submissions-list").innerText = `Error loading submissions: ${err.message}`;
  }
}

fetchSubmissions();
